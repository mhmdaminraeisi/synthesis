SELECT t2.id, t2.rev, t2.content
FROM   (SELECT *
		FROM   t1
		WHERE  TRUE) t2
JOIN   (SELECT id, MIN(rev) AS min_rev
		FROM   (SELECT *
				FROM   t1
				WHERE  TRUE) t3
		GROUP BY id
		HAVING TRUE) t4
ON     (t2.rev = t4.min_rev AND t2.id = t4.id)
